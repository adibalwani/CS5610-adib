This experiment adds comment feature to the existing one. It is a non-persistent one meaning once the page is refreshed the data is vanished. For storage I define an array of JSON 'comments' and any new comments are pushed on this array along with username to be displayed on the table. 

app.controller("CommentController", function ($scope, LoginService, $routeParams) {
    var username = $routeParams.username;
    var comments = [];
    $scope.add = function (text) {
        var obj = { username: username, text: text };
        comments.push(obj);
        $scope.comments = comments;
        $scope.text = null;
    };
});
