About the Experiment

1) Install Karma and Jasmine The following link is used for installation: Karma 2) In the command line inside the app folder enter : karma start Here the browser selected is Chrome. You can change it in karma.conf.js file 3) testing.html file in the app folder is the page for adding two numbers 

describe('AddController', function () {
    beforeEach(module('Add'));
    var $controller;
    beforeEach(inject(function (_$controller_) {
        $controller = _$controller_;
    }));
    describe('Test suite for adding two number', function () {
        var $scope, controller;
        beforeEach(function () {
            $scope = {};
            controller = $controller('AddController', { $scope: $scope });
        });
        it('Add two positive numbers', function () {
            $scope.number1 = 20;
            $scope.number2 = 10;
            $scope.change();
            expect($scope.answer).toEqual(30);
        });
